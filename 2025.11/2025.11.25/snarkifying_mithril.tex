\section{Preliminaries}
\subsection{Notations}

There are some basic notation and conventions used throughout this work.

\begin{itemize}
  \item The security parameter is denoted by $\lambda \in \mathbb{N}$.

  \item For any finite set $S$, we write $|S|$ for its cardinality. We use the notation
        \[
          x \xleftarrow{\$} S
        \]
        to indicate that $x$ is sampled uniformly at random from the finite set $S$.

  \item A function $\mathsf{negl} : \mathbb{N} \to \mathbb{R}_{\ge 0}$ is called \emph{negligible} if for every polynomial $p(\cdot)$ there exists $N \in \mathbb{N}$ such that for all $\lambda \ge N$ it holds that $\mathsf{negl}(\lambda) < 1/p(\lambda)$. We write $\mathsf{negl}(\lambda)$ to denote an unspecified negligible function in the security parameter~$\lambda$.
\end{itemize}


\subsection{BLS Signature Scheme}

Let $\lambda$ be the security parameter. We assume a bilinear group
generation algorithm which, on input $1^\lambda$, outputs
\[
  (q, G_1, G_2, G_T, g_1, g_2, e),
\]
where $G_1,G_2,G_T$ are cyclic groups of prime order $q$, $g_1\in G_1$,
$g_2\in G_2$ are generators, and
$e : G_1 \times G_2 \to G_T$ is a bilinear pairing. Let
$H : \{0,1\}^* \to G_1$ be a hash function modeled as a random oracle.

We write the BLS signature scheme as
\[
  \Pi_{\mathsf{BLS}} = (\mathsf{Setup}, \mathsf{KGen}, \mathsf{Sign}, \mathsf{Ver}).
\]

\begin{itemize}
  \item $pp \leftarrow \mathsf{Setup}(1^\lambda)$:  
        run the group generator to obtain
        $(q, G_1, G_2, G_T, g_1, g_2, e)$ and fix $H$.
        Output public parameters
        \[
          \mathsf{pp} = (q, G_1, G_2, G_T, g_1, g_2, e, H).
        \]

  \item $(sk,vk) \leftarrow \mathsf{KGen}(\mathsf{pp})$:  
        sample a secret key $sk \xleftarrow{\$} \mathbb{Z}_q$, where
        $\mathbb{Z}_q$ denotes the integers modulo $q$, and set
        \[
          vk = g_2^{sk} \in G_2.
        \]
        Output the key pair $(sk, vk)$.

  \item $\sigma \leftarrow \mathsf{Sign}(\mathsf{pp}, sk, msg)$:  
        on input a message $msg \in \{0,1\}^*$, compute
        $h \leftarrow H(msg) \in G_1$ and set
        \[
          \sigma = h^{sk} \in G_1.
        \]
        Output the signature $\sigma$.

  \item $b \leftarrow \mathsf{Ver}(\mathsf{pp}, vk, msg, \sigma)$, where $b \in \{0,1\}$:  
        compute $h \leftarrow H(msg)$ and output
        \[
          b = 1
          \;\text{iff}\;
          e(\sigma, g_2) = e(h, vk).
        \]
\end{itemize}

\subsection{Merkle Tree}

Let $H_{\mathsf{MT}}  : \{0,1\}^* \to \{0,1\}^\lambda$ be a collision-resistant hash function.

We denote the Merkle tree primitive by
\[
  \Pi_{\mathsf{MT}} = (\mathsf{Create}, \mathsf{Check}).
\]

\begin{itemize}

  \item $\mathsf{avk} \leftarrow \mathsf{Create}(\mathbf{vk}, \mathbf{w})$:
        on input a vector of verification keys
        $\mathbf{vk} = (vk_1,\dots,vk_N)$ and a matching stake vector
        $\mathbf{w}=(w_1,\dots,w_N)$, where $N \in \mathbb{N}$ denotes
        the number of parties, set leaves
        $\ell_i := H_{\mathsf{MT}}(vk_i , w_i)$ and output the Merkle root
        $\mathsf{avk}$, which we also call an aggregate verification key.

  \item $b \leftarrow \mathsf{Check}(\mathsf{avk}, vk, w, path)$, where $b \in \{0,1\}$:
        on input a root $\mathsf{avk}$, a pair $(vk,w)$, and an authentication
        path $path$, recompute the root from the leaf
        $\ell := H_{\mathsf{MT}}(vk , w)$ and $path$. Output $b = 1$ iff the
        recomputed root equals $\mathsf{avk}$, and $b = 0$ otherwise.

\end{itemize}


\section{Stake-based threshold multi-signature}


\subsection{Syntax}

We consider a stake-based threshold multi-signature (STMS) scheme whose acceptance is determined on a per-stake basis via a ``$t$ wins in $n$ lotteries'' parametrization, for some integers $t,n \in \mathbb{N}$.



\[
  \Pi_{\mathsf{STMS}}
  =
  (\mathsf{Setup}, \mathsf{KGen}, \mathsf{AKey}, \mathsf{ACheck},
   \mathsf{Sig}, \mathsf{Ver}, \mathsf{ASig}, \mathsf{AVer}) .
\]

\begin{itemize}
  \item $\mathsf{pp} \leftarrow \mathsf{Setup}(1^\lambda)$:  
        on input a security parameter $1^\lambda$, outputs public system parameters $\mathsf{pp}$.  
        These parameters are implicit inputs to all other algorithms.

  \item $(sk,\, vk) \leftarrow \mathsf{KGen}(\mathsf{pp})$:  
        on input the public parameters $\mathsf{pp}$, generates a signing/verification key pair $(sk,vk)$ for a single party.

  \item $\mathsf{avk} \leftarrow \mathsf{AKey}(\mathbf{vk},\, \mathbf{w})$:  
        given a list of verification keys $\mathbf{vk} := (vk_1,\ldots,vk_N)$ and a matching stake vector $\mathbf{w} := (w_1,\ldots,w_N)$,  
        where $N \in \mathbb{N}$ denotes the number of parties, deterministically produces an aggregate verification key $\mathsf{avk}$.

  \item $b \leftarrow \mathsf{ACheck}(\mathsf{avk},\, \mathbf{vk},\, \mathbf{w})$, where $b \in \{0,1\}$:  
        deterministically checks whether $\mathsf{avk}$ is a valid aggregate verification key for $(\mathbf{vk},\mathbf{w})$,  
        outputting $b = 1$ if it accepts and $b = 0$ otherwise.

  \item $\sigma \leftarrow \mathsf{Sig}(vk,\, w,\, sk,\, j,\, m)$:  
        attempts to produce a signature $\sigma$ for message $m$ and lottery index $j \in [n]$  
        under verification key $vk$, secret key $sk$, and stake $w$.  
        It outputs $\bot$ if the user is ineligible for $(m,j)$ under stake $w$.

  \item $b \leftarrow \mathsf{Ver}(vk,\, w,\, j,\, m,\, \sigma)$, where $b \in \{0,1\}$:  
        given a verification key $vk$, a stake value $w$, a lottery index $j$, a message $m$, and a signature $\sigma$,  
        deterministically checks whether $\sigma$ is a valid signature for $(m,j)$ under $(vk,w)$  
        and outputs $b = 1$ if it accepts and $b = 0$ otherwise.

  \item $\tau \leftarrow \mathsf{ASig}(\mathbf{vk},\, \mathbf{w},\, m,\, \{(vk_i,\sigma_i)\}_{i=1}^{t})$:  
        given $(\mathbf{vk},\mathbf{w})$, a message $m$, and a multiset $\{(vk_i,\sigma_i)\}_{i=1}^{t}$ of signatures,  
        attempts to produce an aggregate signature $\tau$; it outputs $\bot$ if the provided inputs are insufficient or inconsistent.

  \item $b \leftarrow \mathsf{AVer}(\mathsf{avk},\, m,\, \tau,\, T)$, where $b \in \{0,1\}$:  
        given a message $m$, an aggregate verification key $\mathsf{avk}$, an aggregate signature $\tau$,  
        and a stake threshold parameter $T$, outputs $b = 1$ if $\tau$ is accepted as a valid aggregate  
        (stake-based threshold) multi-signature on $m$ under $\mathsf{avk}$ and threshold $T$, and $b = 0$ otherwise.

\end{itemize}



\subsubsection{Correctness}

Let 
\[
  \Pi_{\mathsf{STMS}}
  =
  (\mathsf{Setup}, \mathsf{KGen}, \mathsf{AKey}, \mathsf{ACheck},
   \mathsf{Sig}, \mathsf{Ver}, \mathsf{ASig}, \mathsf{AVer})
\]
be a stake-based threshold multi-signature (STMS) scheme as specified above.  
Let $\mathcal{M}$ denote the message space.

\paragraph{Definition (Correctness).}
We say that $\Pi_{\mathsf{STMS}}$ is \emph{correct} if for every security parameter $\lambda \in \mathbb{N}$,
every number of parties $N \in \mathbb{N}$, every vector of verification keys
$\mathbf{vk} = (vk_1,\dots,vk_N)$ generated honestly by $\mathsf{KGen}$, every stake vector
$\mathbf{w} = (w_1,\dots,w_N)$, every message $m \in \mathcal{M}$, every stake threshold parameter
$T$, and every lottery index $j$ in the relevant index set, the following properties hold except with
negligible probability in $\lambda$ over the randomness of all algorithms involved.


\begin{itemize}
  \item \textbf{Correctness of aggregate key.}
        If $\mathsf{avk} \leftarrow \mathsf{AKey}(\mathbf{vk}, \mathbf{w})$, then
        \[
          \mathsf{ACheck}(\mathsf{avk}, \mathbf{vk}, \mathbf{w}) = 1.
        \]

  \item \textbf{Correctness of individual signatures.}
        For every party index $i \in [N]$, let $(sk_i, vk_i)$ be generated by
        $(sk_i, vk_i) \leftarrow \mathsf{KGen}(\mathsf{pp})$.
        If $\sigma_i \leftarrow \mathsf{Sig}(vk_i, w_i, sk_i, j, m)$ and $\sigma_i \neq \bot$, then
        \[
          \mathsf{Ver}(vk_i, w_i, j, m, \sigma_i) = 1.
        \]

  \item \textbf{Correctness of aggregate signatures.}
        Let $t \in \mathbb{N}$ and, for each $i \in [t]$, let $(sk_i, vk_i)$ be generated by
        $(sk_i, vk_i) \leftarrow \mathsf{KGen}(\mathsf{pp})$ and let $w_i$ be the corresponding stake value.
        For a message $m$ and a lottery index $j$, let
        $\sigma_i \leftarrow \mathsf{Sig}(vk_i, w_i, sk_i, j, m)$ be such that $\sigma_i \neq \bot$ for all $i \in [t]$.
        Let
        \[
          \tau \leftarrow \mathsf{ASig}(\mathbf{vk}, \mathbf{w}, m, \{(vk_i,\sigma_i)\}_{i=1}^{t})
          \quad\text{and}\quad
          \mathsf{avk} \leftarrow \mathsf{AKey}(\mathbf{vk}, \mathbf{w}).
        \]
        If the total stake of these $t$ parties meets the threshold, i.e.,
        \[
          \sum_{i=1}^{t} w_i \;\ge\; T,
        \]
        then
        \[
          \mathsf{AVer}(\mathsf{avk}, m, \tau, T) = 1.
        \]
\end{itemize}



\subsubsection{Security}
\paragraph{Stake-based unforgeability $\alpha$-\textsf{UF-CMA}.}
In the unforgeability experiment, the challenger runs $\mathsf{Setup}$, generates honest keys $(sk_i,vk_i)$, fixes $\mathbf{w}$, and gives $(\mathsf{pp},\mathbf{vk},\mathbf{w})$ to the adversary $\mathcal{A}$. The adversary can:
\begin{itemize}
  \item query \emph{signing oracles} for any honest user $i$, any $j$, and any message $m$, receiving $\mathsf{Sig}(vk_i,w_i,sk_i,j,m)$;
  \item \emph{corrupt} users to obtain $sk_i$ (and then act under those keys).
\end{itemize}
Let $C^\ast$ be the set of corrupted users. Let $H^\ast$ be the set of honest users that \emph{participated} on the forged topic (made at least one signing query for the challenge topic) under the forged aggregate key. The experiment \emph{charges} only successful eligibility (i.e., $\mathsf{Sig}\neq\bot$) and enforces the stake bound
\[
  \sum_{i\in C^\ast \cup H^\ast} w_i \;\le\; \alpha\, W .
\]
Finally, $\mathcal{A}$ outputs $(\mathbf{vk}^\ast,\mathsf{avk}^\ast,m^\ast,\tau^\ast)$; it wins if
\[
  \mathsf{ACheck}(\mathsf{avk}^\ast,\mathbf{vk}^\ast,\mathbf{w})=1,\quad
  \mathsf{AVer}(\mathsf{avk}^\ast,m^\ast,\tau^\ast,W)=1,
\]
and $\tau^\ast$ is \emph{fresh}, i.e., it cannot be obtained by aggregating local proofs returned (or derivable) from oracle responses. An STMS is $\alpha$-\textsf{UF-CMA} secure if every PPT adversary wins with only negligible probability.

\paragraph{Strengthened unforgeability $\alpha$-\textsf{UF-CMA-S}.}
As above, but the adversary may first register arbitrary public keys (and corresponding stake placement) before $\mathsf{AKey}$ is run (key-substitution/registration stage). The rest of the game is identical. Security requires negligible winning probability for all PPT adversaries.

\subsection{Instantiation}

\subsubsection{Mithril}

We write the Mithril scheme as
\[
  \Pi_{\mathsf{Mithril}}
  =
  (\mathsf{Setup}, \mathsf{KGen}, \mathsf{AKey}, \mathsf{ACheck}, \mathsf{Sig}, \mathsf{ASig}, \mathsf{AVer}).
\]

\begin{itemize}

  \item $(\mathsf{pp}) \leftarrow \mathsf{Setup}(1^\lambda)$:\\
        run $\Pi_{\mathsf{BLS}}.\mathsf{Setup}(1^\lambda)$ to obtain $pp_{BLS} = (q,G_1,G_2,G_T,g_1,g_2,e,H)$
        
        fix a dense mapping / eligibility function
        \[
        \mathsf{Eval} : ( msg, index, \sigma) \mapsto \{0,1\}^\lambda,
        \]
        % \[
        %   \mathsf{Eval} : G_1 \times \{1,\dots,\mu\} \times \{0,1\}^* \to \{0,1\}^\lambda,
        % \]
        a weighting function
        \[
          \varphi : \mathbb{N} \to [0,1],
        \]
        e.g. $\varphi(w) = 1-(1-f)^w$.

        Index range
        \[
          \mu \in \mathbb{N}, 
        \]

        a quorum or lottery parameter
        \[
          k \in \{1,\dots,\mu\}.
        \]
        Output
        \[
          \mathsf{pp}=(pp_{BLS},\mathsf{Eval},\varphi,\mu,k).
        \]

  \item $(sk,vk) \leftarrow \mathsf{KGen}(\mathsf{pp})$:\\
        parse $pp_{BLS}$ for BLS signature, run $\Pi_{\mathsf{BLS}}.\mathsf{KGen}(\mathsf{pp_{BLS}})$ 
        \[
          (sk,vk) \leftarrow \Pi_{\mathsf{BLS}}.\mathsf{KGen}(\mathsf{pp_{BLS}})
        \]
        Output $(sk,vk)$.

  \item $\mathsf{avk} \leftarrow \mathsf{AKey}(\mathbf{vk},\mathbf{w})$:\\
        given
        \[
          \mathbf{vk}=(vk_1,\dots,vk_N),
        \]
        \[
          \mathbf{w}=(w_1,\dots,w_N),
        \]
        set
        \[
          \mathsf{avk} \leftarrow \Pi_{\mathsf{MT}}.\mathsf{Create}(\mathbf{vk},\mathbf{w}).
        \]

  \item $b \leftarrow \mathsf{ACheck}(\mathsf{avk},\mathbf{vk},\mathbf{w})$, where $b\in\{0,1\}$:\\
        compute
        \[
          \mathsf{avk'} \leftarrow \Pi_{\mathsf{MT}}.\mathsf{Create}(\mathbf{vk},\mathbf{w}).
        \]
        output b=1 iff $\mathsf{avk}=\mathsf{avk'}$. 

  \item $\pi \leftarrow \mathsf{Sig}(\mathsf{pp}, vk, w, sk, j, m)$:\\
        compute the BLS signature
        \[
          \sigma \leftarrow \Pi_{\mathsf{BLS}}.\mathsf{Sign}(\mathsf{pp}, sk, m),
        \]
        compute the eligibility value
        \[
          ev \leftarrow \mathsf{Eval}(\sigma, j, m),
        \]
        and check the threshold
        % \[
        %   ev \;<\; \bigl\lfloor \varphi(w)\cdot 2^\lambda \bigr\rfloor.
        % \]
        \[
          ev \;<\; \bigl\lfloor \varphi(w) \cdot 2^\lambda \bigr\rfloor.
        \]
        If true, output
        \[
          \pi=(j, \sigma, vk, w).
        \]
        Otherwise output $\bot$.




          \item $\tau \leftarrow \mathsf{ASig}(\mathsf{pp}, m, \{\pi_i\}_{i=1}^{t})$:\\
        parse each
        \[
          \pi_i=(j_i,\sigma_i,vk_i,w_i),
        \]
        with pairwise distinct
        \[
          j_i \in \{1,\dots,\mu\}.
        \]
        Set
        \[
          \sigma^\star \;=\; \prod_{i=1}^{t} \sigma_i \;\in\; G_1,
        \]
        \[
          vk^\star \;=\; \prod_{i=1}^{t} vk_i \;\in\; G_2.
        \]
        Derive a batching seed
        \[
          e_\sigma \;\leftarrow\; H\!\bigl(\sigma_1 \,\|\, \cdots \,\|\, \sigma_t \,\bigr),
        \]
        and coefficients for $i=1,\dots,t$
        \[
          \rho_i \;\leftarrow\; \mathsf{H_\lambda}(e_\sigma \,\|\, i) \in \mathbb{Z}_q^\ast.
        \]
        Compute the batch aggregates
        \[
          \mu \;=\; \prod_{i=1}^{t} \sigma_i^{\rho_i} \;\in\; G_1,
        \]
        \[
          ivk \;=\; \prod_{i=1}^{t} vk_i^{\rho_i} \;\in\; G_2.
        \]
        Output
        \[
          \tau \;=\; (\,\sigma^\star,\, vk^\star,\, \mu,\, ivk,\, e_\sigma).
        \]

  \item $b \leftarrow \mathsf{AVer}(\mathsf{pp}, \mathsf{avk}, m, \tau)$, where $b\in\{0,1\}$:\\
        parse
        \[
          \tau=(\sigma^\star, vk^\star, \mu, ivk, e_\sigma).
        \]
        Check that the indices are pairwise distinct.
        Compute
        \[
          h \leftarrow H(m).
        \]
        Verify the equation
        \[
          e(\sigma^\star, g_2) \;=\; e(h, vk^\star).
        \]
        Recompute the batching coefficients for $i=1,\dots,t$
        \[
          \rho_i \;\leftarrow\; \mathsf{H}(e_\sigma \,\|\, i) \in \mathbb{Z}_q^\ast,
        \]
        and the corresponding aggregated public key
        \[
          ivk' \;=\; \prod_{i=1}^{t} vk_i^{\rho_i} \;\in\; G_2.
        \]
        Check consistency of the submitted aggregated key
        \[
          ivk' \;=\; ivk.
        \]
        Verify the batched equation
        \[
          e(\mu, g_2) \;=\; e(h, ivk).
        \]
        Output
        \[
          b \;=\; 1 \text{ iff all checks above hold.}
        \]
\end{itemize}



\subsubsection{Security}

We now formalize the security of a stake-based threshold multi-signature
scheme in the standard game-based style.  Intuitively, an adversary should
not be able to produce an aggregate signature that passes verification
unless all the participating stake either (i) belongs to corrupted parties,
or (ii) comes from honest parties on message--lottery pairs for which the
adversary has obtained valid individual signatures.

\paragraph{Adversarial capabilities.}
We allow the adversary to corrupt an arbitrary subset of users and to obtain
signatures on adaptively chosen message--lottery pairs for honest users.

\paragraph{Experiment.}
Let 
$
  \Pi_{\mathsf{STMS}}
  =
  (\mathsf{Setup}, \mathsf{KGen}, \mathsf{AKey}, \mathsf{ACheck},
   \mathsf{Sig}, \mathsf{Ver}, \mathsf{ASig}, \mathsf{AVer})
$
be an STMS scheme as above, and let $\mathcal{A}$ be a probabilistic
polynomial-time (PPT) adversary.
We define the experiment
\[
  \mathsf{Exp}^{\mathsf{stms\mbox{-}euf}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)
\]
as follows.

\begin{enumerate}
  \item (\emph{Setup.})
        The challenger runs
        $\mathsf{pp} \leftarrow \mathsf{Setup}(1^\lambda)$.
        It fixes a number of users $N \in \mathbb{N}$ (polynomial in $\lambda$)
        and for each $i \in [N]$ samples a key pair
        $(\sk_i,\vk_i) \leftarrow \mathsf{KGen}(\mathsf{pp})$.
        It then samples a stake vector
        $\mathbf{w} := (w_1,\dots,w_N) \in (\mathbb{Z}^+)^N$
        and computes the aggregate verification key
        $\mathsf{avk} \leftarrow \mathsf{AKey}(\mathbf{vk},\mathbf{w})$,
        where $\mathbf{vk} := (\vk_1,\dots,\vk_N)$.
        The challenger gives
        $(\mathsf{pp}, N, \mathbf{vk}, \mathbf{w}, \mathsf{avk})$
        to $\mathcal{A}$.

        It initializes an empty set $\mathcal{C} \subseteq [N]$ of corrupted
        users and an empty set
        $\mathcal{Q} \subseteq [N] \times [n] \times \mathcal{M}$ to record
        signing queries.

  \item (\emph{Oracle access.})
        The adversary $\mathcal{A}$ may make adaptive oracle queries of the
        following types.

        \begin{itemize}
          \item \textbf{Corruption oracle} $\mathsf{Corrupt}(i)$:
                on input an index $i \in [N]$,
                if $i \notin \mathcal{C}$ the challenger adds $i$ to
                $\mathcal{C}$ and returns $\sk_i$ to $\mathcal{A}$.
                (Once a user is corrupted, the adversary can locally compute
                all its signatures and thus no signing oracle is needed for it.)

          \item \textbf{Signing oracle} $\mathsf{Sign}(i,j,m)$:
                on input a user index $i \in [N]$, a lottery index $j \in [n]$,
                and a message $m \in \mathcal{M}$, if $i \notin \mathcal{C}$
                the challenger computes
                \[
                  \sigma \leftarrow
                  \mathsf{Sig}(\mathbf{vk}, \mathbf{w}, \sk_i, j, m)
                \]
                and returns $\sigma$ to $\mathcal{A}$.
                If $\sigma \neq \bot$, the challenger also records
                $(i,j,m)$ in the set $\mathcal{Q}$.
                (If $i \in \mathcal{C}$, the challenger returns $\bot$ since
                $\mathcal{A}$ already knows $\sk_i$.)
        \end{itemize}

        The above interaction may continue for an arbitrary (polynomially
        bounded) number of rounds.

  \item (\emph{Output.})
        Eventually, $\mathcal{A}$ outputs a tuple
        $(m^\ast, \tau^\ast, T^\ast)$, where $m^\ast \in \mathcal{M}$ is a
        message, $\tau^\ast$ is an alleged aggregate signature, and
        $T^\ast \in \mathbb{N}$ is a stake threshold.

        Let $b \leftarrow \mathsf{AVer}(\mathsf{avk}, m^\ast, \tau^\ast, T^\ast)$.
        If $b = 0$, the experiment outputs $0$ (the adversary loses).

        Otherwise, we consider the (implicit) set of signers and lottery
        indices that underlie $\tau^\ast$.  Formally, we assume that there
        exists a uniquely determined multiset
        \[
          \mathsf{Signers}( \tau^\ast )
          \subseteq [N] \times [n]
        \]
        such that, informally, for each $(i,j) \in \mathsf{Signers}(\tau^\ast)$,
        the contribution of user $i$ in lottery $j$ is authenticated inside
        $\tau^\ast$ and is necessary for $\mathsf{AVer}$ to accept.

        The adversary $\mathcal{A}$ is said to \emph{win} if both conditions
        hold:
        \begin{enumerate}
          \item $\mathsf{AVer}(\mathsf{avk}, m^\ast, \tau^\ast, T^\ast) = 1$, and
          \item there exists a pair $(i^\ast,j^\ast) \in \mathsf{Signers}(\tau^\ast)$
                such that $i^\ast \notin \mathcal{C}$ and
                $(i^\ast,j^\ast,m^\ast) \notin \mathcal{Q}$.
        \end{enumerate}

        In this case the experiment outputs $1$; otherwise it outputs $0$.
\end{enumerate}

We define the \emph{stake-based threshold unforgeability advantage} of
$\mathcal{A}$ against $\Pi_{\mathsf{STMS}}$ as
\[
  \mathsf{Adv}^{\mathsf{stms\mbox{-}euf}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)
  :=
  \Pr\big[
    \mathsf{Exp}^{\mathsf{stms\mbox{-}euf}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)
    = 1
  \big],
\]
where the probability is taken over the randomness of the challenger and
$\mathcal{A}$.

\paragraph{Definition (Stake-based threshold unforgeability).}
We say that $\Pi_{\mathsf{STMS}}$ is \emph{stake-based threshold
existentially unforgeable under chosen-message attacks} (STMS-EUF-CMA secure)
if for every PPT adversary $\mathcal{A}$ there exists a negligible function
$\mathsf{negl}(\cdot)$ such that
\[
  \mathsf{Adv}^{\mathsf{stms\mbox{-}euf}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)
  \le \mathsf{negl}(\lambda)
  \quad\text{for all } \lambda \in \mathbb{N}.
\]


\subsubsection{Security}

\paragraph{Stake-based unforgeability $\alpha$-\textsf{UF-CMA}.} \ 
Let 
$
  \Pi_{\mathsf{STMS}}
  =
  (\mathsf{Setup}, \mathsf{KGen}, \mathsf{AKey}, \mathsf{ACheck},
   \mathsf{Sig}, \mathsf{Ver}, \mathsf{ASig}, \mathsf{AVer})
$
be an STMS scheme as specified above, and let
$\mathcal{A}$ be a probabilistic polynomial-time (PPT) adversary.
For a fixed parameter $\alpha \in (0,1)$ we define the following
experiment
\[
  Exp^{\alpha\textsf{-UF-CMA}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)
\]
and say that $\mathcal{A}$ \emph{wins} if the experiment outputs $1$.

\medskip
\noindent\textbf{The unforgeability experiment.}

\begin{enumerate}
  \item \textbf{Initialization.}
        The challenger runs
        $(\mathsf{pp},m) \leftarrow \mathsf{Setup}(1^\lambda)$,
        where $m \in \mathbb{N}$ is the number of lotteries.
        It initializes:
        \begin{itemize}
          \item arrays $(\sk[i])_{i\in\mathbb{N}}$ and $(\vk[i])_{i\in\mathbb{N}}$
                with all entries set to~$\bot$;
          \item three disjoint index sets
                $\mathcal{H},\mathcal{C}_1,\mathcal{C}_2 \subseteq \mathbb{N}$,
                initially all empty, where
                $\mathcal{H}$ denotes honest parties and
                $\mathcal{C}_1,\mathcal{C}_2$ will correspond to parties
                corrupted before and after the first signing query;
          \item a flag $\mathsf{SigningStarted} \gets 0$;
          \item associative arrays

\paragraph{Stake-based unforgeability $\alpha$-\textsf{UF-CMA}.} \ 
\newline
Let $\Pi_{\mathsf{STMS}}
  =
  (\mathsf{Setup}, \mathsf{KGen}, \mathsf{AKey}, \mathsf{ACheck},
   \mathsf{Sig}, \mathsf{Ver}, \mathsf{ASig}, \mathsf{AVer})$ be an STMS scheme as above, and consider the following experiment for an adversary $\mathcal{A}$.
   
\textbf{The unforgeability experiment} $\mathsf{Exp}^{\alpha\text{-UF-CMA}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)$.

\begin{enumerate}
    \item (\emph{Setup.})
    The challenger runs $\mathsf{pp} \leftarrow \mathsf{Setup}(1^\lambda)$, fixes a number of users $N \in \mathbb{N}$, and for each $i \in [N]$ samples a key pair
    $(\sk_i,\vk_i) \leftarrow \mathsf{KGen}(\mathsf{pp})$. 
    
    It then fixes a stake vector  $\mathbf{w} := (w_1,\dots,w_N) \in (\mathbb{Z}^+)^N$ and computes the aggregate verification key $\mathsf{avk} \leftarrow \mathsf{AKey}(\mathbf{vk},\mathbf{w})$, where $\mathbf{vk} := (\vk_1,\dots,\vk_N)$. It also fixes the total number of lottery indices $n \in \mathbb{N}$ and the threshold $k \in \mathbb{N}$. 
    
    The challenger computes the total stake
    \[
      W \ :=\ \sum_{i=1}^N w_i
    \]
    and gives $(\mathsf{pp}, N, \mathbf{vk}, \mathbf{w}, \mathsf{avk}, n, k)$ to $\mathcal{A}$. It then initializes an empty set $\mathcal{C} \subseteq [N]$ of corrupted users and an empty set $\mathcal{Q}$ to record signing queries. %$\mathcal{Q} \subseteq [N] \times [n] \times \mathcal{M}$
    
    \item (\emph{Oracle access.})
    The adversary $\mathcal{A}$ may make corruption and signing oracle queries, adaptively and in any order.
    \begin{itemize}
        \item \textbf{Corruption oracle.}
        On input an index $i \in [N]$, if $i \notin \mathcal{C}$, the challenger adds $i$ to $\mathcal{C}$ and returns $\sk_i$ to $\mathcal{A}$.
        
        \item \textbf{Signing oracle.}
        On input a user index $i \in [N]$, a lottery index $j \in [n]$, and a message $m \in \mathcal{M}$, if $i \notin \mathcal{C}$ the challenger computes
        \[
          \sigma \leftarrow \mathsf{Sig}(\mathbf{vk}, \mathbf{w}, \sk_i, j, m)
        \]
        and returns $\sigma$ to $\mathcal{A}$. If $\sigma \neq \bot$, the challenger records $(i,j,m)$ in the set $\mathcal{Q}$.
        (If $i \in \mathcal{C}$, the challenger simply returns $\bot$ since $\mathcal{A}$ already knows $\sk_i$.)
    \end{itemize}
    
    \item (\emph{Forgery.})
    Eventually, the adversary $\mathcal{A}$ outputs a pair $(m^\ast,\tau^\ast)$, where $m^\ast \in \mathcal{M}$ is a message and $\tau^\ast$ is an alleged aggregate signature.
    
    Let
    \[
      H^\ast \ :=\ \bigl\{\, i \in [N] \setminus \mathcal{C} \;\big|\; \exists j \in [n] \text{ s.t. } (i,j,m^\ast) \in \mathcal{Q} \,\bigr\}
    \]
    be the set of honest users for which $\mathcal{A}$ has obtained individual signatures on $m^\ast$ via the signing oracle, and let
    \[
      C^\ast \ :=\ \mathcal{C}.
    \]
    We define the \emph{effective controlled stake} of $\mathcal{A}$ for $m^\ast$ as
    \[
      W^\ast \ :=\ \sum_{i \in C^\ast \cup H^\ast} w_i.
    \]
    If the stake constraint
    \[
      W^\ast \ \le\ \alpha \, W
    \]
    does not hold, then the experiment outputs $0$ (the adversary loses).
    
    Otherwise, the challenger computes
    \[
      b \leftarrow \mathsf{AVer}(\mathsf{avk}, m^\ast, \tau^\ast, k),
    \]
    where $b \in \{0,1\}$.  The experiment outputs $1$ (the adversary wins) if $b = 1$, and it outputs $0$ otherwise.
\end{enumerate}

We define the adversary's advantage in the above game by
\[
  \mathsf{Adv}^{\alpha\text{-UF-CMA}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)
  :=
  \Pr\Bigl[
    \mathsf{Exp}^{\alpha\text{-UF-CMA}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)
    = 1
  \Bigr],
\]
where the probability is taken over the randomness of the challenger and of $\mathcal{A}$.

\paragraph{Definition (Stake-based unforgeability).}
We say that $\Pi_{\mathsf{STMS}}$ is \emph{$\alpha$-\textsf{UF-CMA} secure} if for every PPT adversary $\mathcal{A}$ there exists a negligible function $\mathsf{negl}(\cdot)$ such that
\[
  \mathsf{Adv}^{\alpha\text{-UF-CMA}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)
  \le \mathsf{negl}(\lambda)
  \quad\text{for all } \lambda \in \mathbb{N}.
\]

We now define the following experiment for a PPT adversary~$\mathcal{A}$.

\medskip
\noindent
\textbf{The unforgeability experiment
$\mathsf{Exp}^{\alpha\textsf{-UF-CMA}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)$}.

\begin{enumerate}
  \item \textbf{Initialization.}
        The challenger runs
        $(\mathsf{pp}, n) \leftarrow \mathsf{Setup}(1^\lambda)$,
        where $n$ denotes the number of lotteries per message and is assumed
        to be part of~$\mathsf{pp}$.
        It initializes arrays of secret and verification keys
        $\mathsf{sk}[i] \gets \bot$, $\mathsf{vk}[i] \gets \bot$ for all
        indices $i$, and sets
        \[
          H \gets \emptyset,\quad
          C_1 \gets \emptyset,\quad
          C_2 \gets \emptyset,\quad
          \mathsf{SigningStarted} \gets 0.
        \]
        It also initializes associative maps
        $I[\cdot,\cdot]$ and $J[\cdot,\cdot]$ and the table
        $\mathsf{Sigs}[\cdot,\cdot,\cdot]$ to the empty set.
        Finally, it sends $(\mathsf{pp}, n)$ to~$\mathcal{A}$.

  \item \textbf{Stake vector.}
        At some point the adversary chooses a stake vector
        $\mathbf{w} = (w_1,\dots,w_N) \in (\mathbb{Z}^+)^N$
        and gives it to the challenger via a special procedure
        $\mathsf{SetW}(\mathbf{w})$ (this procedure may be called only once).
        The challenger records $\mathbf{w}$ and sets
        \[
          w \;\gets\; \sum_{i=1}^{N} w_i
        \]
        (the total stake).

  \item \textbf{Oracle access.}
        The adversary $\mathcal{A}$ is given adaptive oracle access to the
        following procedures.  Whenever a statement of the form
        ``$\textsf{check}~\Phi$'' appears and $\Phi$ is false, the
        experiment outputs $0$ and halts.

        \begin{description}
          \item[$\mathsf{Gen}(i)$:]
                On input an index $i$,
                \[
                  \textsf{check } i \leq |\mathbf{w}|.
                \]
                If $\mathsf{vk}[i] = \bot$ then the challenger runs
                $(\mathsf{sk}[i],\mathsf{vk}[i]) \leftarrow
                   \mathsf{KGen}(\mathsf{pp})$
                and updates
                $H \gets H \cup \{i\}$.
                It returns $\mathsf{vk}[i]$.

          \item[$\mathsf{SetKey}(i,\vk^\ast)$:]
                On input an index $i$ and a verification key $\vk^\ast$,
                \[
                  \textsf{check } i \leq |\mathbf{w}|.
                \]
                The challenger sets
                $\mathsf{vk}[i] \gets \vk^\ast$,
                updates
                $C_1 \gets C_1 \cup \{i\}$ and
                $H \gets H \setminus \{i\}$,
                and returns $\mathsf{ok}$.

          \item[$\mathsf{Corrupt}(i)$:]
                On input an index $i$ the challenger proceeds as follows.
                If $\mathsf{SigningStarted} = 1$ then
                $C_2 \gets C_2 \cup \{i\}$,
                otherwise $C_1 \gets C_1 \cup \{i\}$.
                In both cases it sets $H \gets H \setminus \{i\}$ and
                returns $\mathsf{sk}[i]$ to~$\mathcal{A}$.

          \item[$\mathsf{Sign}(i,m)$:]
                On input an index $i$ and a message $m$,
                \[
                  \textsf{check } i \in H.
                \]
                The challenger sets $\mathsf{SigningStarted} \gets 1$ and,
                for every lottery index $j \in [n]$, computes
                \[
                  \sigma \leftarrow
                    \mathsf{Sig}(\mathbf{vk},\mathbf{w},
                                 \mathsf{sk}[i], j, m).
                \]
                Whenever $\sigma \neq \bot$ it adds the pair
                $(\mathsf{vk}[i],\sigma)$ to the multiset
                $\mathsf{Sigs}[i,m,\mathbf{vk}]$.
                After the loop, it updates the index sets
                \[
                  I[m.\mathsf{topic},\mathbf{vk}]
                    \;\gets\; I[m.\mathsf{topic},\mathbf{vk}] \cup \{i\},
                  \qquad
                  J[m,\mathbf{vk}]
                    \;\gets\; J[m,\mathbf{vk}] \cup \{i\},
                \]
                and returns $\mathsf{Sigs}[i,m,\mathbf{vk}]$.
        \end{description}

        The interaction with these oracles may continue for an arbitrary
        (polynomially bounded) number of rounds.

  \item \textbf{Forgery.}
        Eventually $\mathcal{A}$ outputs a tuple
        $(\mathbf{vk}^\ast,\mathsf{avk}^\ast,m^\ast,\tau^\ast)$,
        where $\mathbf{vk}^\ast = (\vk_1^\ast,\dots,\vk_N^\ast)$ is a vector of
        verification keys, $\mathsf{avk}^\ast$ is an aggregate verification
        key, $m^\ast$ is a message and $\tau^\ast$ is an alleged aggregate
        signature.

        The challenger now performs the following checks
        (this is the procedure $\alpha$-\textsf{UF-CMA} in Game~5).

        \begin{enumerate}
          \item Define the set
                \[
                  C^\ast
                    \;\gets\;
                    C_1 \cup C_2 \cup
                    \{\, i : \vk_i^\ast \neq \mathsf{vk}[i] \,\}.
                \]
          \item Define
                \[
                  H^\ast
                    \;\gets\;
                    \{\, i :
                        i \in I[m^\ast.\mathsf{topic},\mathbf{vk}^\ast]
                        \land \vk_i^\ast = \mathsf{vk}[i]
                     \,\} \cap H.
                \]
          \item \textsf{check}
                \[
                  \sum_{i \in C^\ast \cup H^\ast} w_i \;\le\; \alpha \cdot w.
                \]
          \item \textsf{check}\
                $\mathsf{ACheck}(\mathsf{avk}^\ast,
                                  \mathbf{vk}^\ast,\mathbf{w}) = 1$.
          \item Let
                \[
                  b \;\gets\;
                  \mathsf{AVer}(\mathsf{avk}^\ast,m^\ast,\tau^\ast,T^\ast),
                \]
                where $T^\ast$ is the stake threshold parameter determined by
                the scheme (for example, $T^\ast = \alpha\cdot w$ or is
                encoded in $\mathsf{pp}$).
                The experiment outputs $1$ iff $b=1$, and outputs $0$
                otherwise.
        \end{enumerate}
\end{enumerate}

We define the \emph{$\alpha$-stake-based unforgeability advantage} of
$\mathcal{A}$ against $\Pi_{\mathsf{STMS}}$ as

  :=
  \Pr\!\left[
    \mathsf{Exp}^{\alpha\textsf{-UF-CMA}}_{\Pi_{\mathsf{STMS}},\mathcal{A}}(\lambda)
    = 1
  \right],
\]
where the probability is taken over the randomness of the challenger and of
$\mathcal{A}$.

\section{Security of the Mithril Instantiation}
\label{sec:mithril-security}

In this section we prove that the concrete instantiation of the
stake–based threshold multi-signature (STMS) scheme given in
Appendix~\ref{appendix:instantiation-mithril} (see also
\cite{ChaidosKiayiasMithril2024}) is
$\alpha$-UF-CMA secure assuming the hardness
of the co-CDH problem in the underlying bilinear group and the
knowledge soundness of the SNARK system used for certification.
The instantiation uses BLS signatures as the underlying primitive
and the eligibility function $\Eval$ as defined in
\cite{WuyunSTMSBlueprint2025}:contentReference[oaicite:1]{index=1}.

\begin{theorem}
\label{thm:mithril-ufcma}
Let $\Pi_{\mathsf{Mithril}}$ be the STMS scheme defined in
Appendix~\ref{appendix:instantiation-mithril}.  Assume that:
(i) the BLS signature scheme is existentially unforgeable under chosen
message attacks assuming the co-CDH assumption;
(ii) the hash functions $H$ and $\Eval$ are modelled as random oracles;
(iii) the adversarial stake is at most $\alpha<\frac12$; and
(iv) honest parties erase their one-epoch signing keys immediately after
use. Then $\Pi_{\mathsf{Mithril}}$ satisfies $\alpha$-UF-CMA security.
\end{theorem}

\begin{proof}
Consider an adversary $\mathcal{A}$ that outputs a successful
forgery $(\mathit{vk}^\*,m^\*,\tau^\*)$ in the STMS unforgeability
experiment.  By definition of STMS unforgeability, this means that:
(1) $\ACheck(\avk,\mathit{vk}^\*,w)=1$, 
(2) $\AVer(\avk,m^\*,\tau^\*,T)=1$, and
(3) the total stake of the signers contributing to~$\tau^\*$ but not
queried through corruption or signing oracles is at most $\alpha W$.

\paragraph{Step 1: Reduction to a forged BLS signature.}
By correctness of $\ASig$ (see
Section~\ref{sec:stms-syntax}), if $\AVer$ accepts then the aggregate
signature $\tau^\*$ contains at least $k$ valid individual BLS
signatures $\sigma_i = H(m^\*)^{sk_i}$ coming from \emph{distinct}
verification keys $\mathit{vk}_i$.  Since the adversary controls at
most $\alpha W$ stake and $k$ is chosen as in the Mithril
lottery analysis so that the expected honest contribution exceeds the
adversarial contribution with overwhelming probability, at least one of
these signatures must correspond to an honest user that was neither
corrupted nor queried for a signing oracle on~$m^\*$.

Thus, from $\tau^\*$ we can extract an unqueried, uncorrupted BLS
signature $\sigma^\* = H(m^\*)^{sk^\*}$ under some public key
$\mathit{vk}^\*=g_2^{sk^\*}$.

\paragraph{Step 2: Programming the random oracle.}
The reduction $\mathcal{B}$ simulating the STMS experiment embeds its
co-CDH challenge $(g_1^a,g_2^b)$ into the public key $\mathit{vk}^\*$ by
setting $\mathit{vk}^\* = g_2^b$ and answering all BLS signing,
eligibility, and aggregation queries consistently.  Since $\Eval$ is a
random oracle, $\mathcal{B}$ can ensure that the challenger’s key is
eligible with the correct probability $\varphi(w)$ without revealing the
underlying exponent.

\paragraph{Step 3: Extracting the co-CDH solution.}
When $\mathcal{A}$ outputs its forgery, $\mathcal{B}$ obtains a valid
BLS signature
\[
  \sigma^\* = H(m^\*)^{b}.
\]
Since $\mathcal{B}$ programmed $H(m^\*) = g_1^{a}$ (this happens with
probability at least $1/q_H$ where $q_H$ is the number of hash
queries), we obtain:
\[
  \sigma^\* = (g_1^{a})^{b} = g_1^{ab},
\]
which is a valid solution to the co-CDH problem on input
$(g_1^a,g_2^b)$.

Thus, any PPT adversary forging a Mithril aggregate signature induces a
PPT adversary solving co-CDH with non-negligible probability, violating
assumption~(i).  The only additional failure probabilities arise from:
(i) the probability that all extracted signatures belong to corrupted or
oracle-signing users (bounded by the adversarial stake $\alpha$);
(ii) randomness of the lottery eligibility tests; and
(iii) the random-oracle programming.  All these contribute only
negligible additive terms.

Hence the scheme is $\alpha$-UF-CMA secure.
\end{proof}
