\subsection{Mithril Bridge}


\fcolorbox{gray!20}{gray!5}{
\parbox{0.97\textwidth}{

\paragraph{Setup$_{i,j}()$}
\begin{itemize}
  \item return $\mathsf{Mithril.PGen()}$
\end{itemize}

\paragraph{KGen$_{i,j}(pp_{i,j})$}
\begin{itemize}
  \item return $\mathsf{Mithril.KGen}(pp_{i,j})$
\end{itemize}

\paragraph{RoT$_{i,j}(pp_{i,j}, \mathsf{vks}_{i,j}, C_i, C_j)$}
\begin{itemize}
  \item $[\mathsf{vk}_P : P \in P_{i,j}] \leftarrow \mathsf{parse}(\mathsf{keys}_{i,j})$
  \item $\mathsf{avk} \leftarrow \mathsf{Mithril.Avk}(pp_{i,j}, [\mathsf{vk}_P : P \in P_{i,j}])$
  \item $\mathsf{nonce} \leftarrow \mathsf{Beacon}(C_i)$
  \item put $\mathsf{nonce}$ in $\mathsf{bst}$
  \item get $\mathsf{SD}$ from $C_i$
  \item $\mathsf{rot}_{i,j} \leftarrow (\mathsf{avk}, \mathsf{SD})$
  \item $\mathsf{aux}_{i,j} \leftarrow (\mathsf{rot}_{i,j}, [\mathsf{vk}_P : P \in P_{i,j}])$
  \item return $(\mathsf{rot}_{i,j}, \mathsf{aux}_{i,j})$
\end{itemize}

}}


\fcolorbox{gray!20}{gray!5}{
\parbox{0.97\textwidth}{


\paragraph{Br$_{i,j}(r, m; C_i, C_j; \mathsf{bst})$ when $r = 0$}
\begin{itemize}
  \item get $\mathsf{rot}_{i,j}$ from $\mathsf{bst}$
  \item $\mathsf{code} \leftarrow \mathsf{MITHBridge}[\mathsf{rot}_{i,j}].\mathsf{code}$
  \item $cID \leftarrow \mathsf{Hash}(\mathsf{code})$
  \item $\mathsf{btx}_j \leftarrow (cID, (\mathsf{code}, \mathsf{mem} := \varepsilon))$
  \item return $(\bot, \mathsf{btx}_j, \bot; \mathsf{bst})$
\end{itemize}

\paragraph{Br$_{i,j}(r, m; C_i, C_j; \mathsf{bst})$ when $r \in [1, e-1]$}
\begin{itemize}
  \item return $(\bot, \bot, \bot; \mathsf{bst})$
\end{itemize}

\paragraph{Br$_{i,j}(r, m; C_i, C_j; \mathsf{bst})$ when $r = ke - \Delta - 1$, $k \in \mathbb{N}$}
\begin{itemize}
  \item get $(P, \mathsf{sk})$ from $\mathsf{bst}$
  \item get $pp_{i,j}$ from $\mathsf{bst}$
  \item $(\mathsf{sk}', \mathsf{vk}') \leftarrow \mathsf{Mithril.KGen}(pp_{i,j})$
  \item put $\mathsf{sk}$ in $\mathsf{bst}$
  \item $\sigma' \leftarrow \mathsf{DS.Sig}(\mathsf{sk}, \mathsf{vk}')$
  \item return $(\bot, \bot, (P, \mathsf{vk}', \sigma'); \mathsf{bst})$
\end{itemize}

\paragraph{Br$_{i,j}(r, m; C_i, C_j; \mathsf{bst})$ when $r \in [ke - \Delta, ke - 1]$, $k \in \mathbb{N}$}
\begin{itemize}
  \item put $m$ in $\mathsf{bst}$
  \item return $(\bot, \bot, \bot; \mathsf{bst})$
\end{itemize}

}}

\fcolorbox{gray!20}{gray!5}{
\parbox{0.97\textwidth}{


\paragraph{Br$_{i,j}(r, m; C_i, C_j; \mathsf{bst})$ when $r = ke$, $k \in \mathbb{N}$}
\begin{itemize}
  \item get $(P, \mathsf{sk})$ from $\mathsf{bst}$
  \item get $[\mathsf{vk}'_P : P \in P_{i,j}]$ from $\mathsf{bst}$
  \item $\mathsf{VK}' \leftarrow [\mathsf{vk}'_P : P \in P_{i,j} \wedge \mathsf{DS.Ver}(\mathsf{vk}_P, \mathsf{vk}'_P) = 1]$
  \item $\mathsf{avk}' \leftarrow \mathsf{Mithril.Avk}(pp_{i,j}, \mathsf{VK}')$
  \item get $\mathsf{SD}'$ from $C_i$ at end of round $r = ke$
  \item $\mathsf{cpl} \leftarrow \lfloor r / \Delta_i \rfloor$
  \item $h \leftarrow \mathsf{Hash}(L_i(C_i \text{ up to } \mathsf{cpl}))$
  \item $\mathsf{nonce}' \leftarrow \mathsf{Beacon}(C_i \text{ round } r)$
  \item get $\mathsf{nonce}$ from $\mathsf{bst}$
  \item $\mathsf{msg.topic} \leftarrow r$
  \item $\mathsf{msg.body} \leftarrow (\mathsf{cpl}, h, \mathsf{SD}', \mathsf{avk}', \mathsf{nonce}')$
  \item put $\mathsf{msg}$ in $\mathsf{bst}$
  \item get $\mathsf{SD}$ from $C_i$ for round $(k - 1)e$
  \item compute $w$ using $\mathsf{SD}$
  \item get $w_i$ from $\mathsf{SD}$
  \item $m \leftarrow [\ ]$
  \item for $j^* \in [m]$:
  \begin{itemize}
    \item $\sigma \leftarrow \mathsf{Mithril.Sig}(\mathsf{sk}, j^*, \mathsf{msg}, w_i, w)$
    \item if $\sigma \ne \bot$: $m \leftarrow m : (\mathsf{vk}_P, j^*, \sigma)$
  \end{itemize}
  \item if $m = [\ ]$: $m \leftarrow \bot$
  \item return $(\bot, \bot, m; \mathsf{bst})$
\end{itemize}

\paragraph{Br$_{i,j}(r, m; C_i, C_j; \mathsf{bst})$ when $r \in [ke + 1, ke + \Delta - 1]$, $k \in \mathbb{N}$}
\begin{itemize}
  \item put $m$ in $\mathsf{bst}$
  \item return $(\bot, \bot, \bot; \mathsf{bst})$
\end{itemize}

}}

\fcolorbox{gray!20}{gray!5}{
\parbox{0.97\textwidth}{

\paragraph{Br$_{i,j}(r, m; C_i, C_j; \mathsf{bst})$ when $r = ke + \Delta$, $k \in \mathbb{N}$}
\begin{itemize}
  \item put $m$ in $\mathsf{bst}$
  \item get $\mathsf{msg}$ from $\mathsf{bst}$
  \item get $[\mathsf{vk}_P : P \in P_{i,j}]$ from $\mathsf{bst}$
  \item get $\mathsf{SD}$ from $C_i$ for round $r = ke$
  \item get $\mathsf{VK}$ from $\mathsf{bst}$
  \item form $\mathbf{WVK}$ from $\mathsf{SD}$ and $\mathsf{VK}$
  \item compute $w$ using $\mathsf{SD}$
  \item get $w_i$ from $\mathsf{SD}$
  \item get $T := [(\mathsf{vk}, j^*, \sigma)]$ from $\mathsf{bst}$
  \item $S \leftarrow [(\mathsf{vk}, j^*, \sigma) \in T : \mathsf{Mithril.Ver}(\mathsf{vk}, j^*, \mathsf{msg}, \sigma, w_i, w)]$
  \item $\sigma^* \leftarrow \mathsf{Mithril.ASig}(\mathbf{WVK}, \mathsf{msg}, S)$
  \item return $(\bot, (cID, (\mathsf{msg}, \sigma^*)), \bot; \mathsf{bst})$
\end{itemize}

\paragraph{Br$_{i,j}(r, m; C_i, C_j; \mathsf{bst})$ when $r \in [ke + \Delta + 1, (k+1)e - 1]$, $k \in \mathbb{N}$}
\begin{itemize}
  \item return $(\bot, \bot, \bot; \mathsf{bst})$
\end{itemize}

}}

\fcolorbox{gray!20}{gray!5}{
\parbox{0.97\textwidth}{


\paragraph{GetVerData$_{i,j}(\mathsf{genesis}_j, \mathsf{rot}_{i,j}, L_j)$}
\begin{itemize}
  \item $\mathsf{LS} \leftarrow \mathsf{ledgerState}_j[\mathsf{genesis}_j](L_j)$
  \item $cID \leftarrow \mathsf{Hash}(\mathsf{MITHBridge}[\mathsf{rot}_{i,j}].\mathsf{code})$
  \item $(\_, \mathsf{mem}, \_) \leftarrow \mathsf{LS}[cID]$
  \item return $\mathsf{mem}$
\end{itemize}

\paragraph{$\mathcal{R}^*_{i,j}((\mathsf{cpl}, x, h), C)$}
\begin{itemize}
  \item return $(\mathsf{Header}_i(C, \mathsf{cpl}) = h \land \mathcal{R}_{i,j}(x, L_i(C) \text{ up to } \mathsf{cpl}))$
\end{itemize}

\paragraph{Certify$_{i,j}(\mathsf{rot}_{i,j}, x, C_i)$}
\begin{itemize}
  \item get $\mathsf{crs}_{i,j}$ from $\mathsf{rot}_{i,j}$
  \item $\mathsf{cpl} \leftarrow \lfloor r / \Delta_i \rfloor$
  \item $h \leftarrow \mathsf{Header}_i(C, \mathsf{cpl})$
  \item $\pi \leftarrow \mathsf{Snark.Prove}(\mathsf{crs}_{i,j}, (\mathsf{cpl}, x, h), C)$
  \item return $(\mathsf{cpl}, \pi)$
\end{itemize}

\paragraph{Verify$_{i,j}(\mathsf{mem}, x, \pi)$}
\begin{itemize}
  \item get $\mathsf{crs}_{i,j}$ from $\mathsf{mem}$
  \item get list of $h$ from $\mathsf{mem}$ for the $\mathsf{cpl}$
  \item return $\mathsf{Snark.Verify}(\mathsf{crs}_{i,j}, (\mathsf{cpl}, x, h), \pi)$
\end{itemize}

}}


\fcolorbox{gray!20}{gray!5}{
\parbox{0.97\textwidth}{

\paragraph{Contract MITHBridge[$\mathsf{rot}_{i,j}$]}


\textbf{update}$(\mathsf{msg}, \sigma)$:
\begin{itemize}
  \item get latest $\mathsf{avk}$ from $\mathsf{mem}$
  \item if $\mathsf{avk} = \bot$: get $\mathsf{avk}$ from $\mathsf{rot}_{i,j}$
  \item $(\mathsf{cpl}, h, \mathsf{SD}, \mathsf{avk}, \mathsf{nonce}) \leftarrow \mathsf{msg.body}$
  \item get $w$ from $\mathsf{SD}$
  \item check $\mathsf{Mithril.AVer}(\mathsf{avk}, \mathsf{msg}, \sigma, w)$
  \item $\mathsf{mem} \leftarrow \mathsf{mem} : (\mathsf{cpl}, h, \mathsf{SD}, \mathsf{avk}, \mathsf{nonce})$
\end{itemize}

\textbf{verify}$(x, \pi)$:
\begin{itemize}
  \item if $\mathsf{Verify}_{i,j}(\mathsf{mem}, x, \pi)$: put $(\mathsf{ok}, x, \pi)$ in $\mathsf{mem}$
\end{itemize}

}}

The following are the core algorithms that require zero-knowledge proof in the Mithril-based zero-knowledge proof cross-chain bridge:

\mybox{Mithril Protocol $\Pi_\mathsf{Mithril}$}{white!40}{white!10}{
\textbf{Inputs}: Merkle tree commitment of the committee public keys $\langle AVK \rangle$, aggregate public key $(ivk, ivk_{\mathsf{body}})$, aggregate signature $(\mu, e_\sigma)$, verification information $\{(mvk_i, stake_i, p_i, ev_i, \sigma_i, index_i)\}_{i \in S}$, and message $m$.

\underline{$\mathsf{Verify}(\langle AVK \rangle, ivk, ivk_{\mathsf{body}}, \mu, e_\sigma, \{(mvk_i, stake_i, p_i, ev_i, \sigma_i, index_i)\}_{i \in S}, m, k) \rightarrow 0/1$}:
\begin{itemize}
    \item $ivk = \mathsf{MSP.BKey}(\mathbf{mvk}, e_\sigma)$, and $ivk_{\mathsf{body}} = \mathsf{MSP.AKey}(\mathbf{mvk})$ \quad $\triangleright$ Aggregate key derivation
    \item $(\mu, e_\sigma) = \mathsf{MSP.BSig}(\boldsymbol{\sigma})$ \quad $\triangleright$ Aggregate signature and weight seed derivation
    \item For all $i$: $index_i \leq m$ and for all $i \neq j$: $index_i \neq index_j$ \quad $\triangleright$ Index range and uniqueness checks
    \item For all $i \in \{1, \dots, k\}$: verify that $(mvk_i, stake_i)$ is included in Merkle tree $\langle AVK \rangle$ along path $p_i$ \quad $\triangleright$ Merkle membership check
    \item For all $i \in \{1, \dots, k\}$: verify $\mathsf{MSP.Eval}(topic, index_i, \sigma_i) = ev_i$ \quad $\triangleright$ Eligibility value computation
    \item For all $i \in \{1, \dots, k\}$: verify $ev_i \leq \phi(stake_i)$ \quad $\triangleright$ Weight threshold check
    \item If any of the above checks fail, output $0$; otherwise, output $1$
\end{itemize}
}
